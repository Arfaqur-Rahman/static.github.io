If a=4, b=4, c=2, x1=-0.5+0.5i ,x2=-0.5-0.5i 
			#include<stdio.h>
			#include<math.h>
			int main(){
				float a,b,c,x1,x2,p,q;
				scanf("%f %f %f",&a,&b,&c);
				/*When b^2-4ac is greater than or equal to
				zero, we get two real numbers as the roots of
				the quadratic equation.*/
				if(pow(b,2)-4*a*c>=0){
					x1=(-b + sqrt(pow(b,2)-4*a*c))/(2*a);
					x2=(-b - sqrt(pow(b,2)-4*a*c))/(2*a);
					printf("x1= %f, x2= %f",x1,x2);
				/*When b^2-4ac is less than or equal to
				zero, we get two complex numbers as the roots
				of the quadratic equation. In this case, the
				two roots are x1=p+qi, x2=p-qi, where p= -b/2a
				and q=√(4ac-b^2)/2a*/
				}else{
					p=-b/(2*a);
					q=sqrt(4*a*c-pow(b,2))/(2*a);
					/*Alternatively, you can also put
					q=sqrt( fabs(pow(b,2)-4*a*c) )/(2*a)
					Here fabs() is the absolute value function
					that converts any negative value into a positive
					value.For example, fabs(-2.1)= 2.1  */
					printf("x1= %f+%fi, x2= %f-%fi",p,q,p,q);
				}
				return 0;
			}

			void reverse(int *x,int size);

			int i;
			for(i=0;i<size;i++)
			printf("%d ",*(x+i));
			}

			void reverse(int *x,int size){
				int i=0,j=size-1,t;
				while(i<j){
					t=*(x+i);
					*(x+i)=*(x+j);
					*(x+j)=t;
					i++;
					j--;
				}
			}

			int main(){
			int size,i;
			printf("Enter the size of the array: ");
			scanf("%d",&size);
			int x[size];
			for(i=0;i<size;i++){
				scanf("%d", x+i);
			}
			reverse(x,size);
			show(x,size);
			return 0;
			}
