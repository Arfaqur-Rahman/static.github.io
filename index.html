7. Find the average of three integer numbers, where
num1 = 3
num2 = 5
& num3 = 12
Output will be the average value of these three numbers.
 
			#include<stdio.h>
			int main(){
				int num1=3, num2=5, num3=12;
				float average = (num1+num2+num3)/3.0 ;

3. Write a C program to ask for two integer inputs from the user and calculate their sum, difference, product, quotient and remainder. 
			#include<stdio.h>
			int main(){
				int a, b;
				printf("Enter two integers: ");
				scanf("%d %d", &a, &b);
				printf("The sum of a and b=%d\n", a+b);
				printf("The difference of a and b=%d\n", a-b);
				printf("The product of a and b=%d\n", a*b);
				printf("The quotient of a and b=%f", (float)a/b); /*Here %f is taken because the quotient of a/b may be a floating point number. (float) was written to convert the value of a/b into floating point value.(This is called type casting.) If (float) were not used, a/b would be an integer because a and b are integers themselves.*/
				printf("The remainder of a and b=%d\n", a%b);
				return 0;
			}

7. Take 3 inputs for the arms of a trinagle and check if they form a valid triangle. Then identify the type of the triangle. 
			#include<stdio.h>
			int main(){
				int a, b, c;
				printf("Enter the length of three arms of a triangle: ");
				scanf("%d %d %d",&a,&b,&c);
				if((a+b>c) && (b+c>a) && (c+a>b)){
					printf("It's a valid triangle.\n");
					if(a==b && b==c){
						printf("It's an Equilateral Triangle.\n");
					}else if(a==b  || b==c  || c==a ){

switch()

Problems with switch:

1. Take 3 inputs of 2 integers and 1 operator like this a+b or a-b or a*b or a/b and print the calculation.
If the user types A for +, S for -, M for * or D for /, show the same output. 
			#include<stdio.h>
			int main(){
				int a, b;
				char c;
				printf("Enter 2 values and an operator in this format a+b: ");
				scanf("%d %c %d",&a,&c,&b);
				switch(c){
				case '+':
				case 'A':
					printf("%d+%d=%d",a,b,a+b);
				break;
				case '-':
				case 'S':
					printf("%d-%d=%d",a,b,a-b);
				break;
				case '*':
				case 'M':
					printf("%d*%d=%d",a,b,a*b);
				break;
				case '/':
				case 'D':
					printf("%d/%d=%f",a,b,(float)a/b);
				break;
				default:
					printf("Invalid input");
				}
				return 0;
			}

				case 'u':

				return 0;
			}
